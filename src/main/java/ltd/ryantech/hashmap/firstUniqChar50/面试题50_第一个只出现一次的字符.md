public boolean anagram(String str1, String str2) {
    int[] counter1 = new int[256];
    int[] counter2 = new int[256];
    for (char c : str1.toCharArray()) {
        counter1[c]++;
    }
    for (char c : str2.toCharArray()) {
        counter2[c]++;
    }
    for (int i = 0; i < 256; i++) {
        if (counter1[i] != counter2[i]) {
            return false;
        }
    }
    return true;
}
