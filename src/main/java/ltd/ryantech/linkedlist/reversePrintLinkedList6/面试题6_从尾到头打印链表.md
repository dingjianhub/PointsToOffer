public int[] reversePrint(ListNode head) {
    if (head == null) {
        return new int[0];
    }

    ListNode cur = head;
    Deque<Integer> stack = new LinkedList<>();
    while (cur != null) {
        stack.offerFirst(cur.val);
        cur = cur.next;
    }
    int[] res = new int[stack.size()];
    int idx = 0;
    while (!stack.isEmpty()) {
        res[idx++] = stack.pollFirst();
    }
    return res;
}

    ListNode pre = null;
    ListNode cur = head;
