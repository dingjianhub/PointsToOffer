public ListNode getKthFromEnd(ListNode head, int k) {
    ListNode dummyHead = new ListNode(-1);
    dummyHead.next = head;
    ListNode fast = head;
    ListNode slow = head;
    int count = 0;
    while (count < k && fast != null) {
        fast = fast.next;
        count++;
    }
    while (fast != null) {
        fast = fast.next;
        slow = slow.next;
    }
    return slow;
}
