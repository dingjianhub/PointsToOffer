public int[] reversePrint(ListNode head) {
    if (head == null) {
        return new int[0];
    }

    ListNode newHead = head;
    Stack<ListNode> stack = new Stack<>();
    while (newHead != null) {
        stack.push(newHead);
        newHead = newHead.next;
    }
    int[] res = new int[stack.size()];
    int p = 0;
    while (!stack.isEmpty()) {
        res[p] = stack.pop().val;
        p++;
    }
    return res;
}

    public int[] reversePrint(ListNode head) {
        if (head == null) {
            return new int[0];
        }

        ListNode cur = null;
        ListNode pre = head;
        ListNode t = null;
        int nodesCounter = 0;
        while (pre != null) {
            t = pre.next;
            pre.next = cur;
            cur = pre;
            pre = t;
            nodesCounter++;
        }

        int[] res = new int[nodesCounter];
        ListNode p = cur;
        int i = 0;
        while (p != null) {
            res[i++] = p.val;
            p = p.next;
        }
        return res;
    }
